# 命令

# commander

api文档

node.js命令行界面的完整解决方案。

通过command添加命令,option添加选项,action添加命令回调，最终通过parse解析并执行。

# 添加版本号显示

目标： 输入命令并添加-V或者--version，输出版本号

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .parse(process.argv);  //解析并执行

output:

$ node example.js  //输入
$ 1.0.0            //输出

# 添加命令和响应执行

目标： 添加init命令打印初始化完成。

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init')
    .action(function(){
        console.log('初始化完成')
    })
program.parse(process.argv);  //解析并执行

output:

$ node example.js   //输入
$ 初始化完成         //输出

注意

action后面使用链式调用.parse,需要重起一行通过program

# 给命令添加参数

目标： 给init命令添加一个必填参数打印<参数值>初始化完成。

example.js

var program = require('commander')
//添加一个参数
program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init <name>')
    .action((name,cmd)=>{
        console.log('%s 初始化完成',name)
        console.log('cmd:'+cmd._name)
    })
// 同理可以添加多个必填参数或者是可选参数
// program
//     .version('1.0.0')  //显示版本，默认添加 -V --version选项
//     .command('init <name> <name2> [name3] [name4...]')
//     .action((name,name2,name3,cmd)=>{
//         console.log('%s 初始化完成',name)
//         console.log('%s 初始化完成',name2)
//         console.log('%s 初始化完成',name3)
//         console.log('%s 初始化完成',name4)
//         console.log('cmd:'+cmd._name)
//     })

program.parse(process.argv);  //解析并执行

output:

$ node example.js init vue     //输入
$ vue 初始化完成              //输出
$ cmd:init

注意

<>为必填参数，[]为可选参数，[arg...]为多个可选参数，已数组方式接收，同样适用于arguments和option中，下面会提到。

action的回调函数参数依次获取command参数最后附加返回当前的command上下文

# 给命令添加选项

目标： 给init命令添加选项。

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init <name>')
    .option('-Y --yes [value]','是否使用默认方式初始化')
    .option('-N --no-see','不使用此选项cmd.see为true,否则为false')
    .option('-m --must <value>','给选项添加必选参数')
    .option('-o --must-ption [value]','给选项添加可选参数')
    .option('-d --default [value]','给选项添加可选参数,并设置默认值为[hello]','hello')
    .action((name,cmd)=>{
        console.log('%s 初始化完成',name)
        console.log(`cmd.yes:${cmd.yes}`)
        console.log(`cmd.see:${cmd.see}`)
        console.log(`cmd.must:${cmd.must}`)
        console.log(`cmd.mustOption:${cmd.mustOption}`)
        console.log(`cmd.default:${cmd.default}`)
    })
program.parse(process.argv);  //解析并执行

output:

$ node example.js init name -Y -N -M mu -O -D     //输入
$ name 初始化完成                                //输出
$ cmd.yes:true
$ cmd.see:false
$ cmd.must:mu
$ cmd.mustOption:undefined
$ cmd.default:hello

注意

选项标志可以有多重分割方式，例如-Y --yes、-Y,--yes、-Y|--yes

短标志可以作为单个arg传递，例如-abc相当于-a -b -c。诸如“ --template-engine”之类的多个单词选项是骆驼式的，变成program.templateEngine

以--no开头的多字选项会使得后一个字的布尔值为false。例如，--no-sauce将值设置program.sauce为false,默认不设置此选项则sauce为true。

# 设置该命令允许使用未定义的选项

目标： 默认情况下我们选择不支持的选项会报错error: unknown option，使用allowUnknownOption可以允许命令使用未定义的选项

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init <name>')
    .option('-Y --yes [value]','是否使用默认方式初始化')
    .option('-N --no-see','不使用此选项cmd.see为true,否则为false')
    .option('-m --must <value>','给选项添加必选参数')
    .option('-o --must-ption [value]','给选项添加可选参数')
    .option('-d --default [value]','给选项添加可选参数,并设置默认值为[hello]','hello')
    .allowUnknownOption()
    .action((name,cmd)=>{
        console.log('%s 初始化完成',name)
        console.log(`cmd.yes:${cmd.yes}`)
        console.log(`cmd.see:${cmd.see}`)
        console.log(`cmd.must:${cmd.must}`)
        console.log(`cmd.mustOption:${cmd.mustOption}`)
        console.log(`cmd.default:${cmd.default}`)
    })
program.parse(process.argv);  //解析并执行

output:

$ node example.js init name -f     //输入
$ name 初始化完成                    //输出  
$ md.yes:undefined
$ md.see:true
$ md.must:undefined
$ md.mustOption:undefined
$ md.default:hello

# 给选项添加取值范围

目标： 给init命令选项添加取值范围，只能在范围内赋值，否则赋值无效，依然取默认值。

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init <name>')
    .option('-d --default [value]','取值范围设定只能在选其中一项',/^(你|我|他)$/i,'你')
    .action((name,cmd)=>{
        console.log(`cmd.default:${cmd.default}`)
    })
program.parse(process.argv);  //解析并执行

output:

$ node example.js init hello        //输入
$ cmd.default:你                    //输出

$ node example.js init hello ss     //输入
$ cmd.default:你                    //输出  

$ node example.js init hello 他     //输入
$ cmd.default:他                    //输出

# 给命令添加描述

目标： 给init命令添加描述。

example.js

var program = require('commander')

program

    .description('初始化操作命令')

program.parse(process.argv);  //解析并执行

$ node example.js init --help     //输入
$ Usage: init [options] <name>    //输出
$ 
$ 初始化操作命令
$ 
$ Options:
$   -h, --help  output usage information

# 设置顶级命令使用说明usage

目标： 通过--help查看时，默认为Usage: example [options],我们希望显示为Usage: example 自定义命令使用说明:[command] [option]

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .usage('自定义命令使用说明:[command] [option]')

program.parse(process.argv);  //解析并执行

output:

//输入
$ node example.js init hello       

//输出
$ Usage: example 自定义命令使用说明:[command] [option]   
$ 
$ Options:
$   -V, --version  output the version number
$   -h, --help     output usage information

# 给命令设置别名

目标： 用alias设置init命令笔名为it

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init <name>')
    .alias('it')
    .action((name)=>{
        console.log('%s 初始化完成',name)
    })

program.parse(process.argv);  //解析并执行

output:

$ node example.js it name       //输入
$ name 初始化完成  //输出

# 设置顶级命令和选项

目标： 通过arguments设置顶级命令，同时选项也提升为顶级，通过根下--help可以查看

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .arguments('init <name>')
    .description('init <name> 初始化')
    .option('-d --default [value]','取值范围设定只能在选其中一项',/^(你|我|他)$/i,'你')

program.parse(process.argv);  //解析并执行

output:

$ node example.js --help          //输入
$ Usage: init [options] <name>    //输出
$ 
$ init <name> 初始化
$ 
$ Options:
$   -V, --version         output the version number
$   -d --default [value]  取值范围设定只能在选其中一项 (default: "你")
$   -h, --help            output usage information

# program.on监听命令和选项

目标： 通过program.on 监听顶级命令和顶级选项

example.js

var program = require('commander')

program
    .version('1.0.0')  //显示版本，默认添加 -V --version选项
    .command('init <name>')
    .alias('it')
    .option('-d --default [value]','取值范围设定只能在选其中一项',/^(你|我|他)$/i,'你')

program
    .option('-d --clone')
    
program.on('command:init',function(){
        console.log(this.args)
    })

program.on('option:clone',function(){
    console.log(this.clone)
})

program.parse(process.argv);  //解析并执行

output:

$ node example.js init aa --clone           //输入
$ true                                      //输出
$ [ 'aa' ]

注意

program.on也可以监听命令别名，例如上面也可以写成program.on('command:it')

# 自定义version查看命令选项

目标： 默认是-V --version,我们修改为小写-v --version

var program = require('commander')

program

program.parse(process.argv);  //解析并执行

output:

$ node example.js -v         //输入
$ 1.0.0                     //输出

# 自定义help

目标： 默认是-V --version,我们修改为小写-v --version

example.js

var program = require('commander')

program
    .version('1.0.0','-v --version')  //显示版本，默认添加 -V --version选项

program.on('--help',()=>{
    console.log('\n自定义帮助:')
    console.log('帮助信息')
})

program.parse(process.argv);  //解析并执行

output:

$ node example.js --help         //输入
$ Usage: example [options]      //输出
$ 
$ Options:
$   -v|--version  output the version number
$   -h, --help    output usage information
$ 
$ 自定义帮助:
$ 帮助信息

# 不输入任何参数时默认输出帮助信息

目标： 默认是-V --version,我们修改为小写-v --version

example.js

var program = require('commander')

program
    .version('1.0.0','-v --version')  //显示版本，默认添加 -V --version选项
    .parse(process.argv);

    if(!process.argv.slice(2).length){
        program.help(function(helpTxt){
            //加工处理原帮助信息后返回也可不处理
         return 'outputHelp:\n'+helpTxt
        })
    }
    console.log('结束')

output:

$ node example.js --help         //输入
$ outputHelp:                   //输出
$ Usage: example [options]      
$ 
$ Options:
$   -v|--version  output the version number
$   -h, --help    output usage information
$

注意

还可以使用outputHelp函数，两者的区别是，help执行完会退出当前程序，outputHelp则继续执行，当前例子中如果使用outputHelp追打印出结束两个字
